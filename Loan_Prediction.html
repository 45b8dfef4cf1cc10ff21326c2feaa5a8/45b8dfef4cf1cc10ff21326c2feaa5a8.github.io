import re
# Function to parse the extracted text

def parse_extracted_text(extracted_text):
    data_dict = {}

    # Loan ID
    match_loan_id = re.search(r'Loan_ID\s*(\w+)', extracted_text)
    data_dict['Loan_ID'] = match_loan_id.group(1) if match_loan_id else np.nan

    # Gender
    match_gender = re.search(r'Gender\s*(\w+)', extracted_text)
    data_dict['Gender'] = match_gender.group(1) if match_gender else np.nan

    # Married
    match_married = re.search(r'Married\s*(\w+)', extracted_text)
    data_dict['Married'] = match_married.group(1) if match_married else np.nan

    # Dependents
    match_dependents = re.search(r'Dependents\s*(\w+)', extracted_text)
    data_dict['Dependents'] = '1'#match_dependents.group(1) if match_dependents else np.nan

    # Education
    match_education = re.search(r'Education\s*(\w+)', extracted_text)
    data_dict['Education'] = match_education.group(1) if match_education else np.nan

    # Self Employed
    match_self_employed = re.search(r'Self_Employed\s*(\w+)', extracted_text)
    data_dict['Self_Employed'] = match_self_employed.group(1) if match_self_employed else np.nan

    # Applicant Income
    match_applicant_income = re.search(r'ApplicantIncome\s*(\d+)', extracted_text)
    data_dict['ApplicantIncome'] = int(match_applicant_income.group(1)) if match_applicant_income else np.nan

    # Coapplicant Income
    match_coapplicant_income = re.search(r'CoapplicantIncome\s*(\d+)', extracted_text)
    data_dict['CoapplicantIncome'] = int(match_coapplicant_income.group(1)) if match_coapplicant_income else np.nan

    # Loan Amount
    match_loan_amount = re.search(r'LoanAmount\s*(\d+)', extracted_text)
    data_dict['LoanAmount'] = int(match_loan_amount.group(1)) if match_loan_amount else np.nan

    # Loan Amount Term
    match_loan_amount_term = re.search(r'Loan_Amount_Term\s*(\d+)', extracted_text)
    data_dict['Loan_Amount_Term'] = int(match_loan_amount_term.group(1)) if match_loan_amount_term else np.nan

    # Credit History
    match_credit_history = re.search(r'Credit_History\s*(\d+)', extracted_text)
    data_dict['Credit_History'] = int(match_credit_history.group(1)) if match_credit_history else np.nan

    # Property Area
    match_property_area = re.search(r'Property_Area\s*(\w+)', extracted_text)
    data_dict['Property_Area'] = match_property_area.group(1) if match_property_area else np.nan

    return data_dict

# Function to preprocess the data
def preprocess_data(data_dict, label_encoders, scaler):
    data_df = pd.DataFrame([data_dict])

    # Handle missing values
    imputer = SimpleImputer(strategy='median')
    data_df['LoanAmount'] = imputer.fit_transform(data_df[['LoanAmount']])
    data_df['Loan_Amount_Term'] = imputer.fit_transform(data_df[['Loan_Amount_Term']])
    data_df['Credit_History'] = imputer.fit_transform(data_df[['Credit_History']])

    # Encode categorical variables
    for column in ['Gender', 'Married', 'Dependents', 'Education', 'Self_Employed', 'Property_Area']:
        if column in data_df.columns:
            data_df[column] = label_encoders[column].transform(data_df[column])

    # Feature scaling
    data_df[['ApplicantIncome', 'CoapplicantIncome', 'LoanAmount', 'Loan_Amount_Term']] = scaler.transform(
        data_df[['ApplicantIncome', 'CoapplicantIncome', 'LoanAmount', 'Loan_Amount_Term']]
    )

    # Create additional features
    data_df['Debt_to_Income_Ratio'] = data_df['LoanAmount'] / (data_df['ApplicantIncome'] + data_df['CoapplicantIncome'])
    data_df['Total_Income'] = data_df['ApplicantIncome'] + data_df['CoapplicantIncome']

    # Select relevant features
    features = ['Gender', 'Married', 'Dependents', 'Education', 'Self_Employed', 'ApplicantIncome', 'CoapplicantIncome',
                'LoanAmount', 'Loan_Amount_Term', 'Credit_History', 'Property_Area', 'Debt_to_Income_Ratio', 'Total_Income']

    X_new = data_df[features]

    return X_new

# Model prediction function
def predict_loan_status(image_path, label_encoders, scaler, model):
    extracted_text = extract_text_from_image(image_path)
    data_dict = parse_extracted_text(extracted_text)
    X_new = preprocess_data(data_dict, label_encoders, scaler)
    prediction = model.predict(X_new)
    loan_status = "Approved (Y)" if prediction[0] == 1 else "Rejected (N)"
    return loan_status
